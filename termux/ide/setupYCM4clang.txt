# add plug to .vimrc 
Plug 'ycm-core/YouCompleteMe'

# install plug
vim -c "PlugInstall" -c "q" -c "q"

# check dependecies
cd ~/.vim/plugged/YouCompleteMe
git submodule update --init --recursive

# install

python ./install.py --clang-completer --system-libclang
